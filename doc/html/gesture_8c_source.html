38 /***************************************************************************//**
39  * Takes a given point and determines which region it is in.
40  ******************************************************************************/
41 int8_t getRegion(uint16_t radius, uint16_t angle, uint8_t previous)
42 {
43  // if it is within circle, it is part of the center
44  if (radius < CENTER_RADIUS)
45  return REGION_CENTER;
46  // if already in center, has to past outer circle to leave
47  if (previous == REGION_CENTER && radius < OUTER_CENTER_RADIUS)
48  return -1;
49  // sectors
50  if (angle >= 45 + OVERSHOOT && angle <= 135 - OVERSHOOT)
51  return REGION_UP;
52  if (angle >= 135 + OVERSHOOT && angle <= 225 - OVERSHOOT)
53  return REGION_LEFT;
54  if (angle >= 225 + OVERSHOOT && angle <= 315 - OVERSHOOT)
55  return REGION_DOWN;
56  // for right, check if not in the others
57  if (!(angle >= 45 - OVERSHOOT && angle <= 315 + OVERSHOOT))
58  return REGION_RIGHT;
59  // if not in a sector, then in middle ground
60  return -1;
61 }
